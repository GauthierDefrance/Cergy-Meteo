387 function temperatureToColor(float $temp, float $cold, float $hot) : string {
388 // Définition des températures extrêmes
389
390 // Normalisation de la température entre 0 et 1
391 $normalized = max(0, min(1, ($temp - $cold) / ($hot - $cold)));
392
393 // Déterminer la couleur (dégradé du bleu au rouge en passant par le vert et le jaune)
394 $r = min(255, max(0, 255 * $normalized));
395 $g = min(255, max(0, 255 * (1 - abs($normalized - 0.5) * 2))); // Vert au milieu
396 $b = min(255, max(0, 255 * (1 - $normalized)));
397
398 return sprintf("#%02X%02X%02X", $r, $g, $b);
399 }

409function calculateHumidex($temperature, $humidity) : string {
410 if ($humidity <= 0 || $humidity > 100) return "-";
411
412 // Calcul du point de rosée (Td)
413 $ln_humidity = log($humidity / 100);
414 $td = (237.7 * ($ln_humidity + (7.5 * $temperature) / (237.7 + $temperature))) /
415 (7.5 - $ln_humidity - (7.5 * $temperature) / (237.7 + $temperature));
416
417 // Calcul de la pression de vapeur saturante
418 $e = 6.11 * pow(10, (7.5 * $td / (237.7 + $td)));
419
420 // Calcul de l'humidex
421 $humidex = $temperature + 0.5555 * ($e - 10);
422
423 return round($humidex, 1);
424}

432function calculateWindChill($temperature, $windSpeed) : string {
433 // Windchill s'applique seulement si T ≤ 10°C et vent ≥ 5 km/h
434 if ($temperature > 10 || $windSpeed < 5) return $temperature;
435
436 // Formule de refroidissement éolien
437 $windChill = 13.12 + 0.6215 * $temperature
438 - 11.37 * pow($windSpeed, 0.16)
439 + 0.3965 * $temperature * pow($windSpeed, 0.16);
440
441 return round($windChill, 1);
442}
